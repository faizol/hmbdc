314 void
315 SendTransportEngine::
316 runOnce() HMBDC_RESTRICT {
317  MonoLockFreeBuffer::iterator begin, end;
318  //only START to send when enough sessions are ready to receive or
319  //buffer is full
320  if (hmbdc_likely(!minRecvToStart_
321  || server_->readySessionCount() >= minRecvToStart_)) {
322  minRecvToStart_ = 0; //now started no going back
323  buffer_.peek(begin, end, maxSendBatch_);
324  } else {
325  buffer_.peek(begin, end, 0);
326  }
327 
328  auto it = begin_;
329  uint16_t currentTypeTag = 0;
330  size_t toSendByteSize = 0;
331 
332  while (it != end) {
333  void* ptr = *it;
334  auto item = static_cast<TransportMessageHeader*>(ptr);
335 
336  if (hmbdc_unlikely(!rater_.check(item->wireSize()))) break;
337  if (hmbdc_unlikely(!item->flag
338  && toSendByteSize + item->wireSize() > mtu_)) break;
339  it++;
340  if (hmbdc_unlikely(!currentTypeTag)) {
341  currentTypeTag = item->typeTag();
342  toSendByteSize = item->wireSize();
343  toSend_.push_back(iovec{ptr, item->wireSize()});
344  } else if (item->typeTag() == currentTypeTag) {
345  toSendByteSize += item->wireSize();
346  toSend_.push_back(iovec{ptr, item->wireSize()});
347  } else {
348  server_->queue(currentTypeTag
349  , move(toSend_)
350  , toSendByteSize
351  , it);
352  toSend_.clear();
353 
354  currentTypeTag = item->typeTag();
355  toSendByteSize = item->wireSize();
356  toSend_.push_back(iovec{ptr, item->wireSize()});
357  }
358 
359  if (hmbdc_unlikely(item->flag == app::hasMemoryAttachment::flag)) {
360  auto& a = item->wrapped<app::hasMemoryAttachment>();
361  toSend_.push_back(iovec{a.attachment, a.len});
362  toSendByteSize += a.len;
363  }
364  rater_.commit();
365  }
366 
367  if (toSend_.size()) {
368  server_->queue(currentTypeTag
369  , move(toSend_)
370  , toSendByteSize
371  , it);
372  toSend_.clear();
373  }
374  begin_ = it;
375  //use it here which is the maximum
376  auto newStart = server_->runOnce(begin, it);
377  if (begin != end) { //only need to do this when there r things read from buffer_
378  for (auto it = begin; it != newStart; ++it) {
379  void* ptr = *it;
380  auto item = static_cast<TransportMessageHeader*>(ptr);
381  if (hmbdc_unlikely(item->flag == app::hasMemoryAttachment::flag)) {
382  auto& a = item->wrapped<app::hasMemoryAttachment>();
383  a.release();
384  }
385  }
386  buffer_.wasteAfterPeek(begin, newStart - begin, true);
387  }
388 }
