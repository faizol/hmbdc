213  att->len = bmh.attLen;
214  att->attachment = malloc(sizeof(size_t) + att->len);
215  auto& refCount = *(size_t*)att->attachment;
216  refCount = 1;
217  att->clientData[0] = (uint64_t)&refCount;
218  att->attachment = (char*)att->attachment + sizeof(size_t);
219  att->afterConsumedCleanupFunc = [](app::hasMemoryAttachment* att) {
220  auto& refCount = *((size_t*)att->clientData[0]);
221  if (0 == __atomic_sub_fetch(&refCount, 1, __ATOMIC_RELAXED)) {
