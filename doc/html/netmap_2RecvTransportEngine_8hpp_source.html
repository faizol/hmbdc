159  void recvPackets(struct netmap_ring * HMBDC_RESTRICT ring) HMBDC_RESTRICT {
160  auto cur = ring->cur;
161 
162  while(cur != ring->tail) {
163  struct netmap_slot *slot = &ring->slot[cur];
164  auto *buf = (uint8_t*)NETMAP_BUF(ring, slot->buf_idx);
165  auto* p = reinterpret_cast<hmbdc::comm::eth::pkt*>(buf + virtHeader_
166  - sizeof(hmbdc::comm::eth::virt_header));
167 
168  if (hmbdc_likely(p->ipv4.ip.ip_p == IPPROTO_UDP
169  && p->ipv4.ip.ip_off == htons(IP_DF))) {
170  if (!data_) data_ = (uint8_t*)p->ipv4.body;
171  while (data_ + sizeof(TransportMessageHeader) < buf + slot->len) {
172  auto header = reinterpret_cast<TransportMessageHeader*>(data_);
173  if (data_ + header->wireSize() <= buf + slot->len) {
174  if (subscriptions_.check(header->typeTag())) {
175  auto l = std::min(maxMessageSize_, (size_t)header->messagePayloadLen());
176  outBuffer_.put(header->payload(), l);
177  }
178  data_ += header->wireSize();
179  } else {
180  break;
181  }
182  }
183  }
184  data_ = nullptr;
185  cur = nm_ring_next(ring, cur);
186  }
187 
188  ring->head = ring->cur = cur;
189  }

192  bool verifyChecksum(comm::eth::pkt* HMBDC_RESTRICT packet, size_t payloadWireSize) {
193  {
194  auto tmp = packet->ipv4.ip.ip_sum;
195  packet->ipv4.ip.ip_sum = 0;
196  if (tmp != hmbdc::comm::eth::wrapsum(
197  hmbdc::comm::eth::checksum(&packet->ipv4.ip, sizeof(packet->ipv4.ip), 0))) {
198  return false;
199  }
200  packet->ipv4.ip.ip_sum = tmp;
201  }
202 
203  {
204  auto tmp = packet->ipv4.udp.check;
205  packet->ipv4.udp.check = 0;
206 #pragma GCC diagnostic push
207 #ifdef __clang__
208 #pragma GCC diagnostic ignored "-Waddress-of-packed-member"
209 #endif
210  auto udp = &packet->ipv4.udp;
211  if (tmp != hmbdc::comm::eth::wrapsum(
212  hmbdc::comm::eth::checksum(udp, sizeof(*udp), /* udp header */
213  hmbdc::comm::eth::checksum(packet->ipv4.body, payloadWireSize, /* udp payload */
214  hmbdc::comm::eth::checksum(&packet->ipv4.ip.ip_src, 2 * sizeof(packet->ipv4.ip.ip_src), /* pseudo header */
215  IPPROTO_UDP + (u_int32_t)ntohs(udp->len)))))) {
216  return false;
217  }
218  packet->ipv4.udp.check = tmp;
219  return true;
